function main(data) {
    if (!data || Object.keys(data).length === 0) {
        return {
            result: null
        }
    }

    let items = data?.items
    let resultList = items == null || items.length === 0
        ? null
        : items
            .filter(item =>
                item &&
                item.name &&
                item.regStatus === "存续" &&
                item.matchType !== "历史名称匹配"
            )
            .map(item => item.name)

    return {
        result: resultList && resultList.length > 3
            ? resultList.slice(0, 3)
            : resultList
    }
}

return main(data)
